per_cent = {'ТКБ': 5.6, 'СКБ': 5.9, 'ВТБ': 4.28, 'СБЕР': 4.0}

# Ввод суммы от пользователя
money = float(input("Введите сумму, которую вы планируете положить под проценты: "))

# Расчет накопленных средств за год
deposit = [money * (percent / 100) for percent in per_cent.values()]

print("Суммы накопленных средств по банкам: ", deposit)

# Поиск максимального значения
max_deposit = max(deposit)
max_index = deposit.index(max_deposit)

# Вывод максимальной суммы
print(f"Максимальная сумма, которую вы можете заработать — {max_deposit}")
