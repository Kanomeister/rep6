# Запрашиваем количество билетов
num_tickets = int(input("Введите количество билетов: "))
total_cost = 0

# Запрашиваем возраст для каждого билета и считаем стоимость
for i in range(num_tickets):
    age = int(input(f"Введите возраст посетителя для билета {i + 1}: "))
    
    if age < 18:
        cost = 0  # Бесплатно
    elif 18 <= age <= 25:
        cost = 990  # Стоимость для возраста от 18 до 25 лет
    else:
        cost = 1390  # Полная стоимость для 26 лет и старше
    
    total_cost += cost

# Применяем скидку, если количество билетов больше трех
if num_tickets > 3:
    total_cost *= 0.9  # 10% скидка

# Выводим сумму к оплате
print(f"Сумма к оплате: {total_cost} рублей")
